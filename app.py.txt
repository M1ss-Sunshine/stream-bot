from flask import Flask, request
import requests
from datetime import datetime
import os

app = Flask(__name__)

# Railway Variables
BOT_TOKEN = os.getenv("BOT_TOKEN")
CHAT_ID = os.getenv("CHAT_ID")
CLIENT_ID = os.getenv("CLIENT_ID")
CLIENT_SECRET = os.getenv("CLIENT_SECRET")
STREAMER_LOGIN = os.getenv("STREAMER_LOGIN")

stream_start_time = None
max_viewers = 0


def get_app_token():
    url = "https://id.twitch.tv/oauth2/token"

    params = {
        "client_id": CLIENT_ID,
        "client_secret": CLIENT_SECRET,
        "grant_type": "client_credentials"
    }

    response = requests.post(url, params=params)
    return response.json()["access_token"]


def get_stream_data(token):
    url = f"https://api.twitch.tv/helix/streams?user_login={STREAMER_LOGIN}"

    headers = {
        "Client-ID": CLIENT_ID,
        "Authorization": f"Bearer {token}"
    }

    response = requests.get(url, headers=headers)
    return response.json()


def send_message(text, button=True, image=None):

    if image:
        url = f"https://api.telegram.org/bot{BOT_TOKEN}/sendPhoto"
    else:
        url = f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage"

    keyboard = {
        "inline_keyboard": [[
            {
                "text": "🎮 Смотреть стрим",
                "url": f"https://twitch.tv/{STREAMER_LOGIN}"
            }
        ]]
    } if button else None

    data = {
        "chat_id": CHAT_ID
    }

    if image:
        data["photo"] = image
        data["caption"] = text
    else:
        data["text"] = text

    if keyboard:
        data["reply_markup"] = keyboard

    requests.post(url, json=data)


@app.route("/webhook", methods=["POST"])
def webhook():

    global stream_start_time
    global max_viewers

    data = request.json

    sub_type = data.get("subscription", {}).get("type")

    # Стрим начался
    if sub_type == "stream.online":

        stream_start_time = datetime.utcnow()
        max_viewers = 0

        send_message(
            text="🔴 Я в эфире!\n\nЗалетай на стрим 💜",
            button=True,

            # СЮДА МОЖНО ВСТАВИТЬ ССЫЛКУ НА КАРТИНКУ
            image="https://i.pinimg.com/736x/4f/6f/c8/4f6fc8447d21181bf320a551f64b4fd6.jpg"
        )

    # Стрим закончился
    elif sub_type == "stream.offline":

        if stream_start_time:

            duration = datetime.utcnow() - stream_start_time

            hours = duration.seconds // 3600
            minutes = (duration.seconds % 3600) // 60

            text = (
                f"⚫ Стрим завершён!\n\n"
                f"⏱ Длительность: {hours}ч {minutes}м\n"
                f"👀 Пик зрителей: {max_viewers}\n\n"
                f"Спасибо всем, кто был на стриме 💜"
            )

            send_message(text, button=False)

    # Обновление онлайна
    elif sub_type == "stream.update":

        token = get_app_token()

        stream = get_stream_data(token)

        if stream.get("data"):

            viewers = stream["data"][0]["viewer_count"]

            if viewers > max_viewers:
                max_viewers = viewers

    return "", 200


@app.route("/")
def home():
    return "Bot is running!"


if __name__ == "__main__":
    app.run(
        host="0.0.0.0",
        port=int(os.environ.get("PORT", 3000))
    )
